
create table orders (Id int, Ordrerdate datetime, Orderstatus varchar(20), Noofitems int );
create table customers (Id int, Name varchar(50), Email varchar(100), Phone varchar(15));
create table products (Id int, Name varchar(50), Price decimal(10,2), Stock int);
create table order_items (Id int, OrderId int, ProductId int, Quantity int, Price decimal(10,2));
create table payments (Id int, OrderId int, Amount decimal(10,2), PaymentDate datetime, PaymentMethod varchar(20));
create table shipping (Id int, OrderId int, ShippingAddress varchar(255), ShippingDate datetime, DeliveryDate datetime);
create table reviews (Id int, ProductId int, CustomerId int, Rating int, Comment text, ReviewDate datetime);
create table categories (Id int, Name varchar(50), Description text);
create table product_categories (ProductId int, CategoryId int);
create table discounts (Id int, ProductId int, DiscountPercentage decimal(5,2), StartDate datetime, EndDate datetime);
create table suppliers (Id int, Name varchar(50), ContactEmail varchar(100), ContactPhone varchar(15));
